  <script>
    // Simple cookie banner (no analytics blocking integrado).
    function setLocal(key, val, days=365){ const d=new Date(); d.setTime(d.getTime()+days*24*60*60*1000); document.cookie = key+'='+val+';expires='+d.toUTCString()+';path=/'; }
    function getLocal(key){ const v=document.cookie.match('(^|;)\\s*'+key+'\\s*=\\s*([^;]+)'); return v ? v.pop() : null; }
    function handleCookie(action){
      if(action==='accept'){ setLocal('cloudui_cookies','accepted'); document.getElementById('cookieBanner').style.display='none'; }
      if(action==='decline'){ setLocal('cloudui_cookies','declined'); document.getElementById('cookieBanner').style.display='none'; }
    }
    (function(){
      var pref = getLocal('cloudui_cookies');
      if(!pref) document.getElementById('cookieBanner').style.display='flex';
    })();
  </script>
